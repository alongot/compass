Create src/components/TransferRoadmapView.jsx. The component is self-contained — it fetches its own data via hooks.

**Data derivation:**
```js
const { articulations } = useArticulations(user?.source_institution_id ?? null);
const { institutions } = useInstitutions();
const institution = institutions.find(i => i.id === user?.source_institution_id);
const { mappings: igetcMappings } = useIgetcMappings(institution?.short_name ?? null);

const completedCcCourses = useMemo(
  () => (user?.transcript?.completed ?? []).map(c => c.course),
  [user]
);
const inProgressCcCourses = useMemo(
  () => (user?.transcript?.in_progress ?? []).map(c => c.course),
  [user]
);

const satisfiedIgetcAreas = useMemo(
  () => calculateIgetcProgress(completedCcCourses, igetcMappings),
  [completedCcCourses, igetcMappings]
);

// Also compute in-progress IGETC satisfaction for projection
const projectedIgetcAreas = useMemo(
  () => calculateIgetcProgress([...completedCcCourses, ...inProgressCcCourses], igetcMappings),
  [completedCcCourses, inProgressCcCourses, igetcMappings]
);

const majorRequirements = MAJOR_CONFIGS[user?.target_major_id]?.sections ?? [];
const satisfiedMajorCourses = useMemo(
  () => mapCcCoursesToUcsbRequirements(completedCcCourses, articulations, majorRequirements),
  [completedCcCourses, articulations, majorRequirements]
);

const completedUnits = useMemo(
  () => (user?.transcript?.completed ?? []).reduce((sum, c) => sum + (c.units || 0), 0),
  [user]
);
const inProgressUnits = useMemo(
  () => (user?.transcript?.in_progress ?? []).reduce((sum, c) => sum + (c.units || 0), 0),
  [user]
);
```

**Section 1 — IGETC Areas:**
Heading: "IGETC General Education Requirements"

Build rows array from IGETC_AREAS. For each area:
- Find satisfying CC course: `igetcMappings.find(m => m.igetc_area === area.id && completedCcCourses.includes(m.source_course_code))`
- `satisfied = satisfiedIgetcAreas.has(area.id)`
- `inProgress = !satisfied && projectedIgetcAreas.has(area.id)` (course is in-progress)

Sort rows: satisfied first, then in-progress, then pending — alphabetically by area.id within each group.

Render each row as a horizontal flex row:
- Status indicator: checkmark circle (theme.colors.successActive) if satisfied, half-filled circle (theme.colors.primary with opacity) if in-progress, empty circle (theme.colors.gray[300]) if pending
- Area ID badge (e.g. "1A") in small pill
- Area label
- Right-aligned: satisfying CC course code if satisfied, CC course code + "(in progress)" if in-progress, empty if pending
