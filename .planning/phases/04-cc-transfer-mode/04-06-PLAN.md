**Data derivation:**
```js
const { articulations } = useArticulations(user?.source_institution_id ?? null);

const completedCcCourses = useMemo(
  () => (user?.transcript?.completed ?? []).map(c => c.course),
  [user]
);
const inProgressCcCourses = useMemo(
  () => (user?.transcript?.in_progress ?? []).map(c => c.course),
  [user]
);
const allCcCourses = useMemo(
  () => [...completedCcCourses, ...inProgressCcCourses],
  [completedCcCourses, inProgressCcCourses]
);

const totalUnits = useMemo(
  () => [
    ...(user?.transcript?.completed ?? []),
    ...(user?.transcript?.in_progress ?? []),
  ].reduce((sum, c) => sum + (c.units || 0), 0),
  [user]
);
const unitGapTo60 = Math.max(0, 60 - totalUnits);

// Compute satisfaction for all 12 majors in one useMemo
const majorCards = useMemo(() => {
  return Object.values(MAJOR_CONFIGS).map(major => {
    const requirements = major.sections ?? [];
    const satisfied = mapCcCoursesToUcsbRequirements(allCcCourses, articulations, requirements);
    const totalCourses = requirements.flatMap(s => s.courses ?? []).length;
    const percentSatisfied = totalCourses > 0
      ? Math.round((satisfied.size / totalCourses) * 100)
      : 0;
    return {
      majorId: major.id,
      majorName: major.name,
      isCurrentTarget: major.id === user?.target_major_id,
      satisfiedCount: satisfied.size,
      totalCount: totalCourses,
      percentSatisfied,
      unitGapTo60,
    };
  }).sort((a, b) => {
    // Current target always visible — sort by percent descending
    return b.percentSatisfied - a.percentSatisfied;
  });
}, [allCcCourses, articulations, user, unitGapTo60]);
```
